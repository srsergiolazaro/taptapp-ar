Quick Start

Get Started in Seconds

Just 3 lines of code to unlock powerful AR.

App.tsx 
import { TaptappAR } from '@srsergio/taptapp-ar/react';

export const MyARScene = () => (
  <TaptappAR
    config={{
      targetImageSrc: "https://example.com/target.jpg",
      videoSrc: "https://example.com/overlay.mp4",
      scale: 1.2
    }}
  />
);
app.js 
import { startTracking } from '@srsergio/taptapp-ar';

const tracker = await startTracking({
  targetSrc: './my-image.jpg',  // JIT compiles automatically!
  container: document.getElementById('ar-container'),
  overlay: document.getElementById('overlay'),
  callbacks: {
    onFound: () => console.log("Target Found!"),
    onUpdate: (data) => console.log(data.avgStability)
  }
});
